set

                
(def db {:vendor :mysql :db "db-name"
         :user "user" :password "pass"})

(defseed people
  {:database db :table :people :n 5}
  [:randomized :name :subtype :full-name]
  [:randomized :number :min 50 :max 100])

(seed-table people)

seed
